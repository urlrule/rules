use MyPlace::URLRule::Utils qw/get_url get_url_redirect/;

sub apply_rule {
    my ($url,$rule) = @_;
	my $realurl = $url;
	if($url =~ m/^(.*)\/\?(.+)$/) {
		my $b = $1;
		my $q = $2;
		$q =~ s/ /\+/g;
		$realurl = "$b/search?q=$q";
	}
    return (
		count=>0,
		pass_count=>1,
		pass_data=>[$realurl],
    );
}
